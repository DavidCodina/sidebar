Overview:

This implementation works by restyling <div id="nav-menu" class="navbar-collapse"> such that it becomes a fixed side menu at smaller screen widths. All functionality is independent of bootsrap.js

I couldn't find a CSS-only solution for preventing unwanted transitions on resize, so the following helper was added:


  function throttle(func, timeFrame){
    var lastTime = 0;
    return (
      function(){
        var now = new Date();
        if (now - lastTime >= timeFrame){
          func();
          lastTime = now;
        }
      }
    );
  }


  function removeTransitionsOnResize(element){
    element = element || document.getElementsByTagName('BODY')[0];
    element.classList.add('transition-none');
    setTimeout(function(){ element.classList.remove('transition-none'); }, 500);
  }
  const throttled_removeTransitionsOnResize = throttle(removeTransitionsOnResize, 500);


  window.onresize = function(){ throttled_removeTransitionsOnResize(); }

It works in conjunction with the following CSS rule, originally discussed in CSS Tricks.


  .transition-none * {
    -webkit-transition: none !important;
       -moz-transition: none !important;
        -ms-transition: none !important;
         -o-transition: none !important;
            transition: none !important;
  }
